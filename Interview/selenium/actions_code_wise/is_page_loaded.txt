# check whether page is loaded completely or not

from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.common.exceptions import TimeoutException


def is_page_loaded(self):
  """
  Check if the page is completely loaded.
  :return: True if the page is loaded, False otherwise.
  """
  try:
      all_cards = WebDriverWait(self.driver, 60).until(
      EC.visibility_of_all_elements_located((By.XPATH, "//div[@class='card']"))
      )
      if all_cards:
          print("Page is loaded completely.")
          return True
      return False

    except TimeoutException:
        print("Page did not load within the given time.")
        return False
